local function CreateElement(className, properties)
    local element = Instance.new(className)
    for k, v in pairs(properties or {}) do element[k] = v end
    return element
end

local function CreateGradient(parent, color1, color2, rotation)
    local gradient = Instance.new("UIGradient")
    gradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0, color1), ColorSequenceKeypoint.new(1, color2)}
    gradient.Rotation = rotation or 90
    gradient.Parent = parent
    return gradient
end

local function SpawnFallingStar()
    if not galaxyEnabled or not GalaxyFolder or not LocalPlayer.Character then return end
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local startPos = hrp.Position + Vector3.new(math.random(-800, 800), 600, math.random(-800, 800))
    local endPos = startPos + Vector3.new(math.random(-500, 500), -500, math.random(-500, 500))
    
    local star = CreateElement("Part", {
        Size = Vector3.new(1, 1, 20),
        Position = startPos,
        Color = Colors.RinneRed,
        Material = "Neon",
        Anchored = true,
        CanCollide = false,
        CFrame = CFrame.new(startPos, endPos),
        Parent = GalaxyFolder
    })
    
    TweenService:Create(star, TweenInfo.new(1), {Position = endPos, Transparency = 1}):Play()
    task.delay(1, function() star:Destroy() end)
end

-- Anti-Die Logic Integration
local function SetupAntiDie()
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hum = char:WaitForChild("Humanoid")
    
    hum.BreakJointsOnDeath = false
    hum:SetStateEnabled(Enum.HumanoidStateType.Dead, false)

    local healthConn = hum:GetPropertyChangedSignal("Health"):Connect(function()
        if antiDieEnabled and hum.Health <= 0 then
            hum.Health = hum.MaxHealth
        end
    end)

    local diedConn = hum.Died:Connect(function()
        if antiDieEnabled then
            task.wait()
            local newHum = Instance.new("Humanoid")
            newHum.Name = "ReplacedHumanoid"
            newHum.Parent = char
            workspace.CurrentCamera.CameraSubject = newHum
            hum:Destroy()
        end
    end)
    return {healthConn, diedConn}
end

-- Slider Helper
local function CreateSlider(name, y, min, max, default, callback)
    local label = CreateElement("TextLabel", {Text = name .. ": " .. default, Size = UDim2.new(1, 0, 0, 20), Position = UDim2.new(0, 0, 0, y), BackgroundTransparency = 1, TextColor3 = Colors.White, Font = Enum.Font.GothamBold, TextSize = 10, Parent = ExtraPanel})
    local track = CreateElement("TextButton", {Size = UDim2.new(0.8, 0, 0, 6), Position = UDim2.new(0.1, 0, 0, y + 22), BackgroundColor3 = Color3.fromRGB(50, 50, 50), Text = "", AutoButtonColor = false, Parent = ExtraPanel})
    local fill = CreateElement("Frame", {Size = UDim2.new((default-min)/(max-min), 0, 1, 0), BackgroundColor3 = Colors.Blue, Parent = track})
    local knob = CreateElement("Frame", {Size = UDim2.new(0, 14, 0, 14), Position = UDim2.new((default-min)/(max-min), -7, 0.5, -7), BackgroundColor3 = Colors.White, Parent = track})
    CreateElement("UICorner", {CornerRadius = UDim.new(1, 0), Parent = track}); CreateElement("UICorner", {CornerRadius = UDim.new(1, 0), Parent = fill}); CreateElement("UICorner", {CornerRadius = UDim.new(1, 0), Parent = knob})
    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            local move; move = RunService.RenderStepped:Connect(function()
                local relX = math.clamp((UserInputService:GetMouseLocation().X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
                local val = math.floor(min + (relX * (max - min))); fill.Size = UDim2.new(relX, 0, 1, 0); knob.Position = UDim2.new(relX, -7, 0.5, -7); label.Text = name .. ": " .. val; callback(val)
            end)
            local release; release = UserInputService.InputEnded:Connect(function(input2) if input2.UserInputType == Enum.UserInputType.MouseButton1 or input2.UserInputType == Enum.UserInputType.Touch then move:Disconnect(); release:Disconnect() end end)
        end
    end)
end
